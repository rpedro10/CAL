44  double long1 = node1->getInfo().getCoordinate().getLongitude();
45  double long2 = source->getInfo().getCoordinate().getLongitude();
46  double lat1 = node1->getInfo().getCoordinate().getLatitude();
47  double lat2 = source->getInfo().getCoordinate().getLatitude();
48 
49  double u = sin((lat2 - lat1) / 2);
50  double v = sin((long2 - long1) / 2);
51 
52  return (abs(
53  2.0 * 6371.0 * asin(sqrt(u * u + cos(lat1) * cos(lat2) * v * v))));
